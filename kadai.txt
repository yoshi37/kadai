#include<stdio.h>
int sigma(int a);

void main(void){
	int a,b;
	printf("合計を求めたい数を入力してください：");
	scanf("%d",&a);
	b = sigma(a);	//合計を求める計算式
	printf("1から%dまでの合計は%dです\n",a,b);
}

int sigma(int a){
	int i,b;
	b = 0;
	for(i=1;i<=a;i++){
		if (a==0){
			break;
		}
		b = b + i;
	}
	return b;
}
